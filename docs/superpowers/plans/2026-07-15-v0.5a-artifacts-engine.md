# v0.5a — Artifacts Engine Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Enable the coach to output inline charts and tables in chat via a `render_chart` tool, with a collapsed/expanded SVG card component on the client.

**Architecture:** A new `render_chart` tool returns a structured JSON spec. The client detects `artifact: true` in tool-result parts and renders an `<ArtifactCard>` SVG component (collapsed 80px, expanded 320px). No server-side rendering, no chart libraries, no new tables.

**Tech Stack:** Next.js 16, TypeScript, Zod, Vitest, Tailwind, hand-rolled SVG (same as existing stream charts).

**Spec:** `docs/specs/2026-07-15-v0.5a-artifacts-engine-design.md`

## Global Constraints

- Tool count goes 14 → 15 (one new tool: `render_chart`).
- Tools never throw: this tool always succeeds (it just returns the input as a structured spec).
- TDD: write the failing test first, watch it fail, then implement.

- Chat interface is in `src/components/coach/chat-interface.tsx` — messages render from `m.parts` array. Currently only `type: "text"` parts are displayed; tool-result parts with `artifact: true` need a new code path.

- [ ] **Step 3: Implement the tool**

Create `src/lib/tools/render-chart.ts`:

```ts
import { z } from "zod";
import type { ToolDefinition, ToolContext } from "./registry";

const seriesSchema = z.object({
  label: z.string(),
  data: z
    .array(
      z.object({
        x: z.union([z.string(), z.number()]),
        y: z.number(),
      })
    )
    .min(1),
  color: z
    .string()
    .optional()
    .describe("Tailwind color name or hex. Defaults to theme palette."),
  style: z.enum(["solid", "dashed", "area"]).default("solid"),
});

const parameters = z.object({
  type: z
    .enum(["line", "bar", "area", "table"])
    .describe(
      "Chart type: line (time series), bar (categorical), area (filled), table (rows)."
    ),
  title: z.string().describe("Short chart title shown above the visualization."),
  series: z
    .array(seriesSchema)
    .min(1)
    .describe("Data series to plot."),
  xLabel: z.string().optional().describe("X-axis label."),
  yLabel: z.string().optional().describe("Y-axis label."),
  annotations: z
    .array(
      z.object({
        x: z.union([z.string(), z.number()]),
        label: z.string(),
        color: z.string().optional(),
      })
    )
    .optional()
    .describe("Vertical markers (e.g., race day, today)."),
});

export type ChartSpec = z.infer<typeof parameters>;

async function execute(
  args: z.infer<typeof parameters>,
  _ctx: ToolContext
): Promise<{ artifact: true; chartId: string; spec: ChartSpec }> {
  return {
    artifact: true,
    chartId: crypto.randomUUID(),
    spec: args,
  };
}

export const renderChart: ToolDefinition<typeof parameters> = {
  name: "render_chart",
  description:
    "Render an inline chart or table in the conversation. The athlete sees a " +
    "visual preview that expands on click. Use for trends, comparisons, and " +
    "structured data — anything better shown than described.",
  parameters,
  execute,
};
```

Create `src/components/coach/artifact-card.tsx`:

```tsx
"use client";

import { useState } from "react";
import type { ChartSpec } from "@/lib/tools/render-chart";

const COLORS = ["#34d399", "#38bdf8", "#fbbf24", "#fb7185", "#a78bfa"];

function getColor(idx: number, override?: string): string {
  if (override) return override.startsWith("#") ? override : `var(--color-${override})`;
  return COLORS[idx % COLORS.length];
}

function LineChart({ spec, expanded }: { spec: ChartSpec; expanded: boolean }) {
  const allY = spec.series.flatMap((s) => s.data.map((d) => d.y));
  const minY = Math.min(...allY);
  const maxY = Math.max(...allY);
  const rangeY = maxY - minY || 1;
  const pad = rangeY * 0.1;
  const yMin = minY - pad;
  const yMax = maxY + pad;

  const w = 300;
  const h = expanded ? 200 : 60;

  return (
    <svg viewBox={`0 0 ${w} ${h}`} className="w-full" preserveAspectRatio="none">
      {spec.series.map((s, si) => {
        const points = s.data.map((d, i) => {
          const x = (i / Math.max(s.data.length - 1, 1)) * w;
          const y = h - ((d.y - yMin) / (yMax - yMin)) * h;
          return `${x},${y}`;
        });
        const fill = s.style === "area" || spec.type === "area";
        return (
          <g key={si}>
            {fill && (
              <polygon
                points={`0,${h} ${points.join(" ")} ${w},${h}`}
                fill={getColor(si, s.color)}
                opacity={0.2}
              />
            )}
            <polyline
              points={points.join(" ")}
              fill="none"
              stroke={getColor(si, s.color)}
              strokeWidth={expanded ? 2 : 1.5}
              strokeDasharray={s.style === "dashed" ? "6,3" : undefined}
            />
          </g>
        );
      })}
    </svg>
  );
}

function BarChart({ spec, expanded }: { spec: ChartSpec; expanded: boolean }) {
  const series = spec.series[0];
  if (!series) return null;
  const maxY = Math.max(...series.data.map((d) => d.y), 1);
  const w = 300;
  const h = expanded ? 200 : 60;
  const barW = w / series.data.length * 0.7;
  const gap = w / series.data.length * 0.3;

  return (
    <svg viewBox={`0 0 ${w} ${h}`} className="w-full" preserveAspectRatio="none">
      {series.data.map((d, i) => {
        const barH = (d.y / maxY) * (h - 4);
        const x = i * (barW + gap) + gap / 2;
        return (
          <rect
            key={i}
            x={x}
            y={h - barH}
            width={barW}
            height={barH}
            fill={getColor(0, series.color)}
            rx={2}
          />
        );
      })}
    </svg>
  );
}

function TableView({ spec }: { spec: ChartSpec }) {
  const series = spec.series[0];
  if (!series) return null;
  return (
    <table className="w-full text-xs text-white/80">
      <thead>
        <tr className="border-b border-white/20">
          <th className="px-2 py-1 text-left font-medium">{spec.xLabel || "Item"}</th>
          <th className="px-2 py-1 text-right font-medium">{spec.yLabel || "Value"}</th>
        </tr>
      </thead>
      <tbody>
        {series.data.map((d, i) => (
          <tr key={i} className="border-b border-white/10">
            <td className="px-2 py-1">{d.x}</td>
            <td className="px-2 py-1 text-right font-mono">{d.y}</td>
          </tr>
        ))}
      </tbody>
    </table>
  );
}

export function ArtifactCard({ spec }: { spec: ChartSpec }) {
  const [expanded, setExpanded] = useState(false);

  return (
    <div
      className={`my-2 cursor-pointer overflow-hidden rounded-lg border border-white/20 bg-white/5 transition-all duration-300 ${
        expanded ? "h-auto" : "h-20"
      }`}
      onClick={() => setExpanded(!expanded)}
      role="button"
      aria-expanded={expanded}
      aria-label={`Chart: ${spec.title}`}
    >
      <div className="flex items-center justify-between px-3 pt-2">
        <span className="text-[10px] font-semibold uppercase tracking-wide text-white/60">
          {spec.title}
        </span>
        <span className="text-[9px] text-white/40">
          {expanded ? "▼" : "▶"}
        </span>
      </div>
      <div className={`px-2 pb-2 ${expanded ? "pt-2" : ""}`}>
        {spec.type === "table" ? (
          <TableView spec={spec} />
        ) : spec.type === "bar" ? (
          <BarChart spec={spec} expanded={expanded} />
        ) : (
          <LineChart spec={spec} expanded={expanded} />
        )}
      </div>
      {expanded && spec.series.length > 1 && (
        <div className="flex gap-3 px-3 pb-2">
          {spec.series.map((s, i) => (
            <span key={i} className="flex items-center gap-1 text-[9px] text-white/60">
              <span
                className="inline-block h-2 w-2 rounded-full"
                style={{ backgroundColor: getColor(i, s.color) }}
              />
              {s.label}
            </span>
          ))}
        </div>
      )}
    </div>
  );
}
```
